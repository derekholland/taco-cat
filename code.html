                    // get user input
                    function getValue() {
                      // make alert invisible
                      document.getElementById("alert").classList.add("invisible");
                      // get user string from the page
                      let userString = document.getElementById("userString").value;
                      // check for a palindrome
                      let returnObj = checkForPalindrome(userString);
                      // display our message to the screen
                      displayMessage(returnObj);
                    }
                    
                    // check if string is a palindrome
                    function checkForPalindrome(userString) {
                      //convert to lowercase
                      userString = userString.toLowerCase();
                      // remove spaces and special characters
                      // a-z 0-9
                      let regex = /[^a-z0-9]/gi;
                      // anything that is not a-z 0-9 gets replaced with an empty string
                      userString = userString.replace(regex, "");
                      let reversedString = [];
                      let returnObj = {};
                      // loop through the string backwards and add the reversed string to the revString array
                      for (let i = userString.length - 1; i >= 0; i--) {
                        reversedString += userString[i];
                      }
                      //check if userString equals revString
                      if (userString === reversedString) {
                        // set the msg property
                        returnObj.msg = "Well done! You entered a Palindrome!";
                      } else {
                        returnObj.msg = "Sorry! You did not enter a palindrome!";
                      }
                      // set the reversed property of the returnObj object
                      returnObj.reversed = reversedString;
                      // return the object
                      return returnObj;
                    }
                    
                    function displayMessage(returnObj) {
                      // Display the message
                      document.getElementById("alertHeader").innerHTML = returnObj.msg;
                      // Display the reversed string
                      document.getElementById(
                        "msg"
                      ).innerHTML = `Your phrase reversed is ${returnObj.reversed}`;
                      // remove the invisible display class to reveal the pop up
                      document.getElementById("alert").classList.remove("invisible");
                    }

The code in structured into three functions.

The getValue() Function

This function accepts no arguments. First the "alert" id is targeted with document.getElementById and the CSS class "invisible" is added to the element. This class disables the visibility of the element and this action will be reversed when the displayMessage() function is called. Next we get the string the user typed in by targeting the "userString" id and grabbing the value property. This value is then saved to the userString variable. Now that we have the input value saved to a variable we can pass that variable to the checkForPalindrome() function. This function will return a data object and that data object will be saved to the returnObj variable. Finally, the displayMessage() function is called with the returnObj passed in. This final step will run the function that outputs the results of the program to the screen.

The checkForPalindrome() Function

Using the toLowerCase() method the userString is converted to all lower case characters. This allows us to compare both the string and the reversed string later to determine if they are equal. Next a regular expression is created and assigned to the variable regex This regular expression is used to remove any spaces or special characters which is important for accurately comparing strings. Next using the replace method anything that is not a-z or 0-9 will be replaced with an empty string.

Next the variable reversedString is created and assigned an empty array. This array will be used to store the string we reverse in the for loop. Another variable is created and assigned an empty object. Now using a for loop we run through userString in reversed order and add the letters of the string on each iteration. This fills the empty array created earlier with a reversed version of the user string. Now we can check if the user string is a palindrome. Using an if statement we check if userString equals reversedString. If they are equal then we have found a palindrome. For a match we set the msg property of the returnObj object to a message telling the user the word is a palindrome. If no match the msg property is set to a message that reflects this. We exit the if code block and set the reversed property of the returnObj object to the reversedString variable. This will be used to show the user the reversed word in the displayMessage() function.

The displayMessage() Function

The function takes the returnObj object as an argument and displays the msg and reversedString properties in the form of a message to the user on the screen. To do this we set the inner HTML of the element with the "alertHeader" id to the msg property of the returnObj object. Next we do the same thing except target the element with the "msg" class. The inner HTML for this element is set to a template string with the reversed property nested. The template string tells the user what the string reversed is. Finally, to get this message to show up on the screen we have to remove the invisible class that was previously set in the getValue() function by grabbing the element with the alert id and remving the "invisible" class from the classList. Now the results of the palindrome check are showing on the screen!
